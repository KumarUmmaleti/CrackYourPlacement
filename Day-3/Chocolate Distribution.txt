class Solution
{
    public long findMinDiff (ArrayList<Integer> a, int n, int m)
    {
        // your code here
        int mindiff=Integer.MAX_VALUE;
        Collections.sort(a);

        for(int i=m-1;i<a.size();i++)
        {
            mindiff=Math.min((a.get(i)-a.get(i-(m-1))),mindiff);
        }
        return mindiff;
        
    }
}
