class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        int p = 0;
        int b = prices[0];

        for(int i=1;i<n;i++){
            if(prices[i]<b){
                b=prices[i];
            }
            else if(prices[i]-b>p){
                p = prices[i]-b;
            }
        }
        return p;
    }
}
