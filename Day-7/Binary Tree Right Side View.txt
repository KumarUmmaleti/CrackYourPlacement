class Solution {
    public List<Integer> rightSideView(TreeNode root) {
        List<Integer> res = new ArrayList<>();
        view(res, root, 0);
        return res;
    }
    public void view(List<Integer> arr, TreeNode node, int level){
        if(node==null){
            return;
        }
        if(level==arr.size()){
            arr.add(node.val);
        }
        else{
            arr.set(level, node.val);
        }
        view(arr, node.left, level+1);
        view(arr, node.right, level+1);
    }
}
