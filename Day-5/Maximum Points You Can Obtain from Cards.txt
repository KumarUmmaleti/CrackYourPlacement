class Solution {
    public int maxScore(int[] cardPoints, int k) {
        int n = cardPoints.length;
        int sumleft = 0;
        int sumright = 0;
        int maxsum = 0;

        for(int i=0;i<k;i++){
            sumleft = sumleft + cardPoints[i];
        }
        maxsum = Math.max(maxsum,sumleft);
        int rightidx = n-1;
        for(int i=k-1;i>=0;i--){
            sumleft = sumleft-cardPoints[i];
            sumright = sumright + cardPoints[rightidx];
            rightidx = rightidx-1;
            maxsum = Math.max(maxsum,sumleft+sumright);
        }
        return maxsum;       
    }
}
