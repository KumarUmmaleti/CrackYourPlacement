class Solution {
    public int maxArea(int[] height) {
        int n = height.length;
        int i = 0;
        int j = n-1;
        int area = 0;
        int res = 0;


        while(i<j){
            area = (j-i)*Math.min(height[i],height[j]);
            res = Math.max(res,area);
            if(height[i]<height[j]){
                i++;
            }
            else{
                j--;
            }
        }
        return res;
    }
}
